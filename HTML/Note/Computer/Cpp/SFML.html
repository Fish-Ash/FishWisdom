            int main()
            {
                RenderWindow window(VideoMode(1200, 800), "SFML works!");
                Color background_color = Color::White;

                while (window.isOpen())
                {
                    Event event;
                    while (window.pollEvent(event))
                    {
                        if (event.type == Event::Closed)
                            window.close();
                    }

                    window.clear(background_color);
                    window.display();
                }

                return 0;
            }
